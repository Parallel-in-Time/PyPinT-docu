\begin{frame}
  \frametitle{Deferred Corrections \hspace{2em} \onslide<2>{ and \hspace{2em}  Integral Deferred Corrections}}

 \framesubtitle{\scriptsize\normalfont  \alt<1>{ V. Pereyra, “On improving an approximate solution of a functional equation by deferred corrections,” Numer. Math., vol. 391, no. 044, pp. 376–391, 1966.}
  					{ A. Dutt, L. Greengard, and V. Rokhlin, “Spectral deferred correction methods for ordinary differential equations,” BIT Numer. Math., vol. 40, no. 2, pp. 241–266, Jun. 2000.}}
  \begin{columns}[T]
    \begin{column}{7cm}
      $ y'(t) = f\left( t, y\left( t \right) \right),\quad y\left( 0 \right)= y_{0} $\\[1.5em]
    Using the residual
\\[1em]
	$r(t) = f\left( t, \tilde y\left( t \right) \right) - \tilde y'\left( t \right)$
\\[1em]
    to compute the error
    \begin{align*}
	e'(t) &= f(t,\tilde y + e) - f\left( t, \tilde y \right) + r'\left( t \right)\\
	e(0)  &= 0
    \end{align*}
    for the next update
    \begin{align*}
      \tilde y_{j+1} = \tilde y_{j} + e_{j}
    \end{align*}
    

    \end{column}
    \begin{column}{7cm}

      \onslide<2>{
      $   y(t) = y_0 + \int_{0}^{t}  f\left( \tau, y\left( \tau \right) \right) \mathrm{d}\tau, \quad y\left( 0 \right)= y_{0} $ \\[1.5em]
      Using the residual\\[1em]
      $r(t) = y_{0} - \tilde y\left( t \right) + \int_{0}^{t} f\left(\tau, y\left( \tau \right) \right)  \mathrm{d}\tau $\\[1em]
    to compute the error
    \begin{align*}
      e(t) = \int_{0}^{t}f(\tau,\tilde y + e) - f\left( \tau, \tilde y \right) \mathrm{d}\tau + r\left( t \right)
    \end{align*}\\[0.5em]
    for the next update
    \begin{align*}
      \tilde y_{j+1} = \tilde y_{j} + e_{j}
    \end{align*}
	}

  \end{column}
\end{columns}


\end{frame}

\begin{frame}
  \frametitle{ Spectral Deferred Corrections }
  \framesubtitle{\scriptsize\normalfont A. Dutt, L. Greengard, and V. Rokhlin, “Spectral deferred correction methods for ordinary differential equations,” BIT Numer. Math., vol. 40, no. 2, pp. 241–266, Jun. 2000.}
  \begin{center}
 \includegraphics[width=6cm]{src/sdc_time_domain} 
  \end{center}

One sweep is performed by
\begin{align*}
  u_{n+1}^{k+1} = u_{n}^{k+1} + \Delta t \left[ F\left( u_{n+1}^{k+1} \right) - F\left(u_{n+1}^{k}  \right)) \right] +\mathcal{I}_{n}^{n+1}\left( F\left( \mathbf{u}^{k} \right) \right) 
\end{align*}
new main constituent is the quadrature operator
\begin{align*}
  \mathcal{I}_{n}^{n+1} \mathbf{y} = \sum_{i=1}^{m} \omega_{i} \tilde y_i \approx \int_{t_n}^{t_{n+1}} y\left( \tau \right) \mathrm{d} \tau
\end{align*}

What is needed? \textcolor{fzjyellow}{Quadrature},  \textcolor{fzjyellow}{Evaluation of $F$} (, \color{fzjyellow}{Space Solver}) $\longrightarrow$ \textcolor{fzjyellow}{new Integrator}

\end{frame}

\begin{frame}
  \frametitle{Multilevel SDC}
  \framesubtitle{\scriptsize\normalfont R. Speck, D. Ruprecht, and M. Emmett, “A multi-level spectral deferred correction method,” arXiv Prepr. arXiv …, 2013.}
\center Slightly different sweep
\begin{align*}
  u_{n+1}^{k+1} = u_{n}^{k+1} + \Delta t \left[ F\left( u_{n+1}^{k+1} \right) - F\left(u_{n+1}^{k}  \right)) \right] +\mathcal{I}_{n}^{n+1}\left( F\left( \mathbf{u}^{k} \right) \right) + \textcolor{fzjblue30}{\tau_{n}^{k}}
\end{align*}
\center and multiple level, correction $\textcolor{fzjblue30}{\tau_{n}^{k}}$ based on information from different levels
\\[1.5em]
\begin{columns}[T]

	  \item \textcolor{fzjyellow}{Interpolation} (space and time)
	  \item \textcolor{fzjyellow}{Restriction} (space and time)
	\end{itemize}
  \end{column}

    \item<7-> open-source hosted on \includegraphics[height=1em]{src/GitHub-Mark-32px.png}~GitHub\\
      {\onslide<8->\color{fzjblue50}$\Rightarrow$ reaching young \& rising developers and scientists promoting PinT algorithms\\[1.5em]}
  \end{itemize}

\begin{frame}[fragile]
  \frametitle{Implementation of SDC and MLSDC}
  \framesubtitle{Python Script for a SDC run with 3 Gauss-Lobatto nodes}
  \begin{lstlisting}[language=Python]
problem = LambdaU(complex(-1.0, 1.0))

comm = ForwardSendingMessaging()
solver = Sdc(communicator=comm)

quadr = QuadratureBase(nodes=GaussLobatto(3), weight=Polynomial(1))

solver.init(quadr, problem)

solution = solver.run(integrator=SemiImplicitSdc)
  \end{lstlisting}
